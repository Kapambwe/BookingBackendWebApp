@code {
    private List<Project> projects = new();
    private List<Project> filteredProjects = new();
    private RadzenDataGrid<Project> grid;

    private string selectedStatus;
    private string selectedType;
    private string searchTerm;
    private bool showFilters = false;

    private int totalProjects = 0;
    private int activeProjects = 0;
    private int totalFarmersEnrolled = 0;
    private decimal averageEnrollmentRate = 0;
    private decimal budgetUtilization = 0;

    private readonly List<string> statusOptions = Enum.GetNames(typeof(ProjectStatus)).ToList();
    private readonly List<string> typeOptions = Enum.GetNames(typeof(ProjectType)).ToList();

    protected override async Task OnInitializedAsync()
    {
        await LoadData();
    }

    private async Task LoadData()
    {
        projects = await ProjectService.GetProjectsAsync();
        CalculateStatistics();
        ApplyFilters();
    }

    private void CalculateStatistics()
    {
        totalProjects = projects.Count;
        activeProjects = projects.Count(p => p.Status == ProjectStatus.Active);
        totalFarmersEnrolled = projects.Sum(p => p.EnrolledFarmers);
        averageEnrollmentRate = projects.Any() ? projects.Average(p => p.TargetFarmers > 0 ? (decimal)p.EnrolledFarmers / p.TargetFarmers * 100 : 0) : 0;
        budgetUtilization = projects.Any() ? projects.Average(p => p.TotalBudget > 0 ? (p.FundsUtilized / p.TotalBudget) * 100 : 0) : 0;
    }

    private void ApplyFilters()
    {
        filteredProjects = projects.AsEnumerable().ToList();

        if (!string.IsNullOrEmpty(selectedStatus))
            filteredProjects = filteredProjects.Where(p => p.Status.ToString() == selectedStatus).ToList();

        if (!string.IsNullOrEmpty(selectedType))
            filteredProjects = filteredProjects.Where(p => p.Type.ToString() == selectedType).ToList();

        if (!string.IsNullOrEmpty(searchTerm))
            filteredProjects = filteredProjects.Where(p =>
                p.ProjectName.Contains(searchTerm, StringComparison.OrdinalIgnoreCase) ||
                p.ProjectCode.Contains(searchTerm, StringComparison.OrdinalIgnoreCase) ||
                p.Description.Contains(searchTerm, StringComparison.OrdinalIgnoreCase)).ToList();

        filteredProjects = filteredProjects.ToList();
    }

    private void OnFilterChanged()
    {
        ApplyFilters();
        StateHasChanged();
    }

    private void OnSearchChanged()
    {
        ApplyFilters();
        StateHasChanged();
    }

    private decimal CalculateOverallProgress(Project project)
    {
        var metrics = new[]
        {
            project.TargetFarmers > 0 ? (decimal)project.EnrolledFarmers / project.TargetFarmers * 100 : 0,
            project.TargetLandArea > 0 ? project.ActualLandArea / project.TargetLandArea * 100 : 0,
            project.TargetYield > 0 ? project.CurrentYield / project.TargetYield * 100 : 0,
            project.TargetIncomeIncrease > 0 ? project.CurrentIncomeIncrease / project.TargetIncomeIncrease * 100 : 0,
            project.TargetAdoptionRate > 0 ? (decimal)project.CurrentAdoptionRate / project.TargetAdoptionRate * 100 : 0
        };

        return metrics.Average();
    }

    private async void ShowCreateProjectDialog()
    {
        await DialogService.OpenAsync<CreateProjectDialog>("Create New Project",
            new Dictionary<string, object>(),
            new DialogOptions() { Width = "800px", Height = "700px", Resizable = true, Draggable = true });

        await LoadData();
    }

    private async void AssignFarmers(Project project)
    {
        await DialogService.OpenAsync<AssignFarmersDialog>("Assign Farmers to Project",
            new Dictionary<string, object> { { "Project", project } },
            new DialogOptions() { Width = "900px", Height = "700px", Resizable = true, Draggable = true });

        await LoadData();
    }

    private async void ViewProgress(Project project)
    {
        await DialogService.OpenAsync<ProjectProgressDialog>("Project Progress",
            new Dictionary<string, object> { { "ProjectId", project.Id } },
            new DialogOptions() { Width = "1000px", Height = "800px", Resizable = true, Draggable = true });
    }

    private async void EditProject(Project project)
    {
        await DialogService.OpenAsync<EditProjectDialog>("Edit Project",
            new Dictionary<string, object> { { "Project", project } },
            new DialogOptions() { Width = "800px", Height = "700px", Resizable = true, Draggable = true });

        await LoadData();
    }

    private async Task DeleteProject(Project project)
    {
        var result = await DialogService.Confirm($"Are you sure you want to delete project {project.ProjectName}?", "Delete Project",
            new ConfirmOptions() { OkButtonText = "Yes", CancelButtonText = "No" });

        if (result == true)
        {
            var success = await ProjectService.DeleteProjectAsync(project.Id);
            if (success)
            {
                NotificationService.Notify(NotificationSeverity.Success, "Success", "Project deleted successfully");
                await LoadData();
            }
            else
            {
                NotificationService.Notify(NotificationSeverity.Error, "Error", "Failed to delete project");
            }
        }
    }

    private string GetTotalProjects()
    {
        return totalProjects.ToString("N0");
    }

    private string GetActiveProjects()
    {
        return activeProjects.ToString("N0");
    }

    private string GetTotalFarmersEnrolled()
    {
        return totalFarmersEnrolled.ToString("N0");
    }

    private BadgeStyle GetProjectStatusBadgeStyle(ProjectStatus status)
    {
        return status switch
        {
            ProjectStatus.Active => BadgeStyle.Success,
            ProjectStatus.Completed => BadgeStyle.Info,
            ProjectStatus.Planning => BadgeStyle.Warning,
            _ => BadgeStyle.Secondary
        };
    }

    private string GetProjectTimeline(Project project)
    {
        return $"{project.StartDate.ToString("MMM yyyy")} - {project.EndDate.ToString("MMM yyyy")}";
    }

    private string GetDaysRemaining(Project project)
    {
        var days = (project.EndDate - DateTime.Now).Days;
        return days > 0 ? $"{days} days remaining" : "Overdue";
    }

    private string GetEnrollmentProgress(Project project)
    {
        return $"{project.EnrolledFarmers} / {project.TargetFarmers}";
    }

    private BadgeStyle GetOverallProgressBadgeStyle(decimal progress)
    {
        if (progress >= 80) return BadgeStyle.Success;
        if (progress >= 50) return BadgeStyle.Warning;
        return BadgeStyle.Primary;
    }
}
